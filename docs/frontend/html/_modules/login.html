        try:
            p = self.hashPass(p)

                        "type": "login",
                        "uname": u,
                        "passwd": p
                    }).encode()))
            k = self.session_manager.get()
            self.session_manager.task_done()

                self.pNotSameLabel.configure(text="Invalid Username/Password")

            elif k["id"] == 1:
                self.rotate = False
                self.cook = k["sessionId"]
                self.cUname = u
                self.explode()

        except socket.error as err:
            print(err)

[docs]    def rsa_keys(self, usrname):
        publicKey, privateKey = rsa.newkeys(2048)
        privfile = open(f".{usrname}priv.pem", "x")
        privfile.write(privateKey.save_pkcs1().decode("utf-8"))
        privfile.close()
        return publicKey



[docs]    def hashPass(self, p):
        hasher = nacl.hash.sha512
        digest = base64.b64encode(
            hasher(p.encode(), encoder=nacl.encoding.HexEncoder)).decode()
        return digest
