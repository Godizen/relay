[docs]    def handle(self) -> None:
        """The threaded TCP request handler. Handles all requests from clients.
        Uses if and elif blocks to determine the request.
        """

        sk: PrivateKey = PrivateKey.generate()
        self.request.sendall(sk.public_key._public_key)
        pkc: PublicKey = PublicKey(self.request.recv(4096))

        b = Box(sk, pkc)

        while True:
            k: str = self.request.recv(4096)
            k = b.decrypt(k).decode()
            print(k)
            data: "dict[str, Any]" = json.loads(k)

            if data["type"] == "login":
                data["passwd"] = base64.b64decode(
                    data["passwd"].encode()).decode()
                cursor.execute(
                    """SELECT UNAME FROM Users WHERE (UNAME=%s AND PASSWD=%s)""",
                    (data["uname"], data["passwd"]),
                )
                if cursor.fetchone():
                    session_id = secrets.token_urlsafe(64)
                    online_list_add_queue.put(
                        User(
                            uname=data["uname"],
                            session_id=session_id,
                            connection=self.request,
                            address=self.client_address,
                        ))
                    self.request.sendall(
                        b.encrypt(
                            json.dumps({
                                "type": "login",
                                "id": 1,
                                "sessionId": session_id
                            }).encode()))
                else:
                    self.request.sendall(
                        b.encrypt(
                            json.dumps({
                                "type": "login",
                                "id": 0
                            }).encode()))

            elif data["type"] == "register":
                cursor.execute("""SELECT UNAME FROM Users WHERE (UNAME=%s)""",
                               (data["uname"], ))

                if cursor.fetchone():
                    self.request.sendall(
                        b.encrypt(
                            json.dumps({
                                "type": "register",
                                "id": -1
                            }).encode()))
                else:
                    session_id = secrets.token_urlsafe()
                    cursor.execute(
                        """INSERT INTO Users (UNAME, PASSWD) VALUES (%s, %s)""",
                        (data["uname"], data["passwd"]),
                    )
                    cursor.execute(
                        "INSERT INTO Keys (UNAME, PUB_KEYS) VALUES (%s, %s);",
                        (data["uname"], data["pub"]),
                    )
                    online_list_add_queue.put(
                        User(
                            uname=data["uname"],
                            session_id=session_id,
                            connection=self.request,
                            address=self.client_address,
                        ))
                    self.request.sendall(
                        b.encrypt(
                            json.dumps({
                                "type": "register",
                                "id": 1,
                                "sessionId": session_id
                            }).encode()))

            elif data["type"] == "get_online":
                u: "list[str]" = []
                for user in online_list:
                    u.append(user.uname)
                self.request.sendall(
                    b.encrypt(
                        json.dumps({
                            "type": "get_online",
                            "id": 1,
                            "online": u
                        }).encode()))

            elif data["type"] == "get_key":
                cursor.execute(
                    """SELECT PUB_KEYS FROM Keys WHERE (UNAME=%s)""",
                    (data["uname"], ))
                self.request.sendall(
                    b.encrypt(
                        json.dumps({
                            "type": "get_key",
                            "id": 1,
                            "key": cursor.fetchone()[0],  # type:ignore
                        }).encode()))

            elif data["type"] == "logout":
                try:
                    work_list = online_list.copy()
                    for user in work_list:
                        if user.session_id == data["sessionId"]:
                            online_list_remove_queue.put(user)
                            break
                    else:
                        self.request.sendall(
                            b.encrypt(
                                json.dumps({
                                    "type": "logout",
                                    "id": 0
                                }).encode()))
                    self.request.sendall(
                        b.encrypt(
                            json.dumps({
                                "type": "logout",
                                "id": 1
                            }).encode()))
                except Exception as err:
